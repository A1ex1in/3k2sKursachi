CREATE DATABASE "банк"
    WITH
    OWNER = postgres
    ENCODING = 'UTF8'
    LC_COLLATE = 'Russian_Russia.1251'
    LC_CTYPE = 'Russian_Russia.1251'
    LOCALE_PROVIDER = 'libc'
    TABLESPACE = pg_default
    CONNECTION LIMIT = -1
    IS_TEMPLATE = False;
-------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
-------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
-------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
BEGIN;
	CREATE TABLE PUBLIC.валюта
	(
		ид_валюта bigint PRIMARY KEY, название text
	);
	CREATE TABLE PUBLIC.тип_операции
	(
		ид_тип_операции bigint PRIMARY KEY, название text
	);
	CREATE TABLE PUBLIC.состояние_счет
	(
		ид_состояние_счет bigint PRIMARY KEY, состояние text
	);
	CREATE TABLE PUBLIC.состояние_заявка
	(
		ид_состояние_заявка bigint PRIMARY KEY, состояние text
	);
	CREATE TABLE PUBLIC.филлиал
	(
		ид_филлиал bigint PRIMARY KEY, название text, страна text, город text, улица text, дом bigint, почтовый_индекс bigint
	);
	CREATE TABLE PUBLIC.клиент
	(
		ид_клиент bigint PRIMARY KEY, ф text, и text, о text, паспорт bigint, телефон bigint, страна text, город text, улица text, дом bigint
	);
	CREATE TABLE PUBLIC.тип_счет
	(
		ид_тип_счет bigint PRIMARY KEY, ид_валюта bigint REFERENCES PUBLIC.валюта(ид_валюта), название text, процет bigint
	);
	CREATE TABLE PUBLIC.заявка
	(
		ид_заявка bigint PRIMARY KEY, ид_клиент bigint REFERENCES PUBLIC.клиент(ид_клиент), ид_состояние_заявка bigint REFERENCES PUBLIC.состояние_заявка(ид_состояние_заявка), уровень_дохода bigint, сумма bigint
	);
	CREATE TABLE PUBLIC.счет
	(
		ид_счет bigint PRIMARY KEY, ид_тип_счет bigint REFERENCES PUBLIC.тип_счет(ид_тип_счет), ид_состояние_счет bigint REFERENCES PUBLIC.состояние_счет(ид_состояние_счет), 
		ид_филлиал bigint REFERENCES PUBLIC.филлиал(ид_филлиал), ид_заявка bigint REFERENCES PUBLIC.заявка(ид_заявка), сумма bigint
	);
	CREATE TABLE PUBLIC.операция_счет
	(
		ид_операция_счет bigint PRIMARY KEY, ид_счет bigint REFERENCES PUBLIC.счет(ид_счет), ид_тип_операции bigint REFERENCES PUBLIC.тип_операции(ид_тип_операции), сумма bigint, дата DATE
	);
END;
